    firebase.initializeApp(firebaseConfig);
    const db = firebase.database();
    let isAdmin = false;

    // Karte & Icons
    const map = L.map('map', { crs: L.CRS.Simple, minZoom: -3, maxZoom: 2 });
    const bounds = [[0, 0], [8192, 8192]];
    L.imageOverlay('https://gta-5-map.com/tiles/cl_0_0.png', bounds).addTo(map);
    map.fitBounds(bounds);

    const houseIcon = L.icon({
        iconUrl: 'https://cdn-icons-png.flaticon.com/512/619/619153.png',
        iconSize: [30, 30], popupAnchor: [0, -15]
    });

    // Admin Toggle
    function toggleAdmin() {
        const pw = prompt("Passwort:");
        if (pw === "admin12347") { // Dein Admin-Passwort
            isAdmin = true;
            document.getElementById('adminControls').style.display = 'flex';
            alert("Admin aktiviert");
        }
    }

    // DATEN AUS FIREBASE LADEN (Echtzeit)
    db.ref('markers').on('value', (snapshot) => {
        // Karte leeren (außer Hintergrund)
        map.eachLayer((layer) => {
            if (layer instanceof L.Marker) map.removeLayer(layer);
        });

        const data = snapshot.val();
        for (let id in data) {
            const m = data[id];
            const options = m.category === "Anwesen" ? { icon: houseIcon } : {};
            const marker = L.marker(m.coords, options).addTo(map);

            let content = `<b>${m.name}</b><br>Typ: ${m.category}`;
            if (isAdmin) {
                content += `<br><button onclick="deleteMarker('${id}')">Löschen</button>`;
            }
            marker.bindPopup(content);
        }
    });

    // PUNKT SPEICHERN
    map.on('click', (e) => {
        if (!isAdmin) return;
        const name = prompt("Name des Ortes:");
        if (name) {
            const category = document.getElementById('category').value;
            db.ref('markers').push({
                name: name,
                category: category,
                coords: [e.latlng.lat, e.latlng.lng]
            });
        }
    });

    // PUNKT LÖSCHEN
    function deleteMarker(id) {
        if(confirm("Löschen?")) {
            db.ref('markers/' + id).remove();
        }
    }
